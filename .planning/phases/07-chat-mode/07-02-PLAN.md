    **5. 轮次分组逻辑（从 events 提取 round）**

    在 App.tsx 内部使用 `useMemo` 构建轮次数据：

    ```tsx
    type RoundDisplay = {
      round_num: number;
      userMessage: UserMessageEvent | null;
      agentEvents: Event[];
      toolCalls: ToolCallInfo[];
      tokenUsage: TokenUsage | null;
    };

    const rounds = useMemo<RoundDisplay[]>(() => {
      if (!activeSessionId) return [];
      const events = eventsBySession[activeSessionId] ?? [];
      const toolCalls = toolCallsBySession[activeSessionId] ?? [];
      
      // Group events by round using user_message as boundary
      const result: RoundDisplay[] = [];
      let currentRound: Event[] = [];
      let currentUserMsg: UserMessageEvent | null = null;
      
      for (const event of events) {
        if (event.event_type === "user_message") {
          // Save previous round
          if (currentUserMsg || currentRound.length > 0) {
            result.push(buildRound(result.length + 1, currentUserMsg, currentRound, toolCalls));
          }
          currentUserMsg = event as UserMessageEvent;
          currentRound = [];
        } else if (event.event_type === "round_end") {
          // Finalize current round
          currentRound.push(event);
          result.push(buildRound(result.length + 1, currentUserMsg, currentRound, toolCalls));
          currentUserMsg = null;
          currentRound = [];
        } else {
          currentRound.push(event);
        }
      }
      
      // Flush remaining events (active round, no round_end yet)
      if (currentUserMsg || currentRound.length > 0) {
        result.push(buildRound(result.length + 1, currentUserMsg, currentRound, toolCalls));
      }
      
      return result;
    }, [eventsBySession, activeSessionId, toolCallsBySession]);
    ```

    `buildRound` 辅助函数：
    ```tsx
    function buildRound(
      roundNum: number,
      userMsg: UserMessageEvent | null,
      events: Event[],
      toolCalls: ToolCallInfo[]
    ): RoundDisplay {
      // Extract token usage from step_end events
      let cumulativeTokenUsage: TokenUsage | null = null;
      let promptTokens = 0, completionTokens = 0;
      for (const e of events) {
        if (e.event_type === "step_end" && e.token_usage) {
          promptTokens += e.token_usage.prompt_tokens;
          completionTokens += e.token_usage.completion_tokens;
        }
      }
      if (promptTokens > 0 || completionTokens > 0) {
        cumulativeTokenUsage = { prompt_tokens: promptTokens, completion_tokens: completionTokens, total_tokens: promptTokens + completionTokens };
      }
      
      // Get tool call IDs referenced in this round's events
      const toolCallIds = new Set(
        events.filter(e => e.event_type === "tool_call_start")
          .map(e => (e as ToolCallStartEvent).tool_call_id)
      );
      const roundToolCalls = toolCalls.filter(tc => toolCallIds.has(tc.tool_call_id));
      
      return {
        round_num: roundNum,
        userMessage: userMsg,
        agentEvents: events,
        toolCalls: roundToolCalls,
        tokenUsage: cumulativeTokenUsage,
      };
    }
    ```

    需要 import `ToolCallStartEvent` from `@/lib/eventTypes`（或使用 `(e as any).tool_call_id` 避免导入，为简单起见使用类型断言）。
